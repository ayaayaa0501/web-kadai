拡散アニメーション作品

  • 拡散するアニメーション
  • 内トロコイドを使用し徐々に拡大される
  • ランダムで色が変化（寒色系）
  • デザインの美しさを主に作成

float A6 = -5.0;//振幅
float t;//右
float t0;//右
float t1;//経過時間
float rad1 = 0.00001; //スピード//右
float rad2 = 0.00000001; //右

float rad3 = 0.3; //スピード
float rad5 = 0.008;
float scale1 =1.0;//右
float scale2 =1.0;
float rad = TWO_PI/180.0;

void setup() {
size(400, 400);
background(0);
noStroke();
}

void draw() {
//右（分散）
x = (A1 - A2)*cos(t) +A2*cos(((A1 - A2)/A2)*t);
y = (A1 -A2)*sin(t) - A2*sin(((A1 - A2)/A2)*t);
fill(random(0,100),random(100,255),random(200,255));
rect(x*scale1 + width/2+ width/200, y*scale1 + height/2, 1, 1);
t += rad; //時間を進める
scale1 += 0.0015; //徐々に拡大
t += rad1; //時間を進める
scale1 += 0.0025; //徐々に拡大
A2 -= rad2; //A2が0未満になったら増減を反転させる
if (A2 < 3.5) rad2 = -rad2;
}
