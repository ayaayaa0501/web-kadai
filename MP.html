ノイズ

  • ノイズを使ったプログラミング作品
  • 図形は線のみを使用。色彩はグレースケールのみ。
  • クリックごとにグラフィックが更新されるプログラム
  • デザインの美しさを主に作成

float angle = 0.0;
float myNoise = 0.0; //ノイズの初期値
float start = 0.0; //myNoiseのスタート地点
float step = 0.01; //ステップの数値が少ないと変化が滑らかになる

void setup() {
size(400, 400);
stroke(255,200);
}

void draw() {
background(0);
translate(width/2, height/2); //座標系を画面中央に移動
myNoise = start; //myNoiseをスタート地点で初期化
for (int x = 0; x < 200 ; x += 1) {
//ノイズを使ってy座標を設定
float y = noise(myNoise)*380;
rotate(radians(30)); //座標系を回転
line(x/1.5, 0, 1.5, y);
//ランダムを使った場合は連続性がない
//line(x, 0, x, random(height));
myNoise += step; //ノイズの値を更新
angle += 0.5;
if (angle >=360.0) angle = 0.0;
}
}

void mousePressed() {
start = random(10); //myNoiseのスタート位置を変える
}
